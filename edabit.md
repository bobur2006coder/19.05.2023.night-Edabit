function drop(arr, val = 1) {
	let arrtwo = [];
	for (let i=val; i<arr.length; i++){
		arr[i]>val

	}
	return arrtwo

### [5.Reverse an Array](https://edabit.com/challenge/kJQYTCCWSnzhXG9dn)

### Answer

```js
function reverse(arr) {
	let arrtwo = [];
	for (let i =0; i<arr.length; i++){
		arrtwo.push(arr[i])
	}
	return arrtwo.reverse()
}

```
